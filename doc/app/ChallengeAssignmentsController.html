     # File app/controllers/challenge_assignments_controller.rb, line 176
176:   def update_multiple
177:     @errors = []
178:     params.each_pair do |key, val|
179:       action, id = key.split(/_/)
180:       next unless %(approve default undefault cover).include?(action)
181:       assignment = ChallengeAssignment.where(:id => id).first
182:       unless assignment
183:         @errors << ts("Couldn't find assignment with id #{id}!")
184:         next
185:       end
186:       case action
187:       when "default"
188:         # default_assignment_id = y/n
189:         assignment.default || @errors << ts("We couldn't default the assignment for #{assignment.offer_byline}")
190:       when "undefault"
191:         # undefault_[assignment_id] = y/n - if set, undefault
192:         assignment.defaulted_at = nil
193:         assignment.save || @errors << ts("We couldn't undefault the assignment covering #{assignment.request_byline}.")
194:       when "approve"
195:         assignment.get_collection_item.approve_by_collection if assignment.get_collection_item
196:       when "cover"
197:         # cover_[assignment_id] = pinch hitter pseud
198:         next if val.blank? || assignment.pinch_hitter.try(:byline) == val
199:         pseud = Pseud.parse_byline(val).first
200:         if pseud.nil?
201:           @errors << ts("We couldn't find the user #{val} to assign that to.")
202:         else
203:           assignment.cover(pseud) || @errors << ts("We couldn't assign #{val} to cover #{assignment.request_byline}.")
204:         end
205:       end
206:     end
207:     if @errors.empty?
208:       flash[:notice] = "Assignment updates complete!"
209:       redirect_to collection_assignments_path(@collection)
210:     else
211:       flash[:error] = @errors
212:       redirect_to collection_assignments_path(@collection)
213:     end 
214:   end
