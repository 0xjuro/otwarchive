get_id(path, user, classname) click to toggle source

We try and get ids to use in paths here with a bit of custom guesswork The fallback is just using the first object of a given class 

     # File app/controllers/home_controller.rb, line 81
 81:   def get_id(path, user, classname)
 82:     begin    
 83:       if classname == "person"
 84:         return People.all.first.to_param
 85:       end
 86:       
 87:       object = case classname
 88:       when "user"
 89:         user || User.first
 90:       when "pseud"
 91:         user.try(:default_pseud) || Pseud.first
 92:       when "medium"
 93:         Media.first
 94:       when "collection"
 95:         user.owned_collections.first
 96:       when "restricted_work"
 97:         Work.where(:restricted => true).in_collection(Collection.find(@last_id)).first
 98:       when "tag_wrangler"
 99:         if permit?("tag_wrangler")
100:           user
101:         else
102:           Role.find_by_name("tag_wrangler").users.first
103:         end
104:       when "nomination"
105:         TagSetNomination.for_tag_set(OwnedTagSet.find(@last_id)).owned_by(user).first
106:       when "setting"
107:         AdminSetting.first
108:       when "assignment", "claim", "signup"
109:         klass = "challenge_#{classname}".classify.constantize
110:         query = classname == "assignment" ? klass.by_offering_user(user) :
111:                   (classname == "signup" ? klass.by_user(user) :  
112:                     (classname == "claim" ? klass.by_claiming_user(user) : nil))
113:         if path.match('collection')
114:           query = query.in_collection(Collection.find(@last_id))
115:         end
116:         query.first
117:       when "item", "participant"
118:         "collection_#{classname}".classify.constantize.where(:collection_id => @last_id).first
119:       when "tag_wrangling", "user_creation", "translator", "translation"
120:         # not real objects
121:         nil
122:       else
123:         klass = classname.classify.constantize
124:         if klass.column_names.include?("user_id")
125:           klass.where(:user_id => user.id).first
126:         elsif klass.column_names.include?("pseud_id")
127:           klass.where(:pseud_id => user.default_pseud.id).first
128:         else
129:           classname.classify.constantize.first
130:         end
131:       end
132: 
133:       # return the id
134:       object.try(:id)
135:     rescue
136:       @errors << "NOT SURE HOW TO FIND #{classname}"
137:       nil
138:     end      
139:   end
