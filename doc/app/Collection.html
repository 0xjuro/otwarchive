386:   def self.sorted_and_filtered(sort, filters, page)
387:     pagination_args = {:page => page}
388: 
389:     # build up the query with scopes based on the options the user specifies
390:     query = Collection.top_level
391:     
392:     if !filters[:title].blank?
393:       # we get the matching collections out of autocomplete and use their ids
394:       ids = Collection.autocomplete_lookup(:search_param => filters[:title], 
395:                 :autocomplete_prefix => (filters[:closed].blank? ? "autocomplete_collection_all" : (filters[:closed] ? "autocomplete_collection_closed" : "autocomplete_collection_open"))
396:              ).map {|result| Collection.id_from_autocomplete(result)}
397:       query = query.where("collections.id in (?)", ids)
398:     else
399:       query = (filters[:closed] == "true" ? query.closed : query.not_closed) if !filters[:closed].blank?
400:     end
401:     query = (filters[:moderated] == "true" ? query.moderated : query.unmoderated) if !filters[:moderated].blank?
402:     if sort.match /item_count/
403:       query = query.with_item_count
404:       pagination_args.merge!({:order => sort})
405:     else
406:       query = query.order(sort)
407:     end
408: 
409:     if !filters[:fandom].blank?
410:       fandom = Fandom.find_by_name(filters[:fandom])
411:       if fandom
412:         (fandom.approved_collections & query).paginate(pagination_args)
413:       else
414:         []
415:       end
416:     else
417:       query.paginate(pagination_args)
418:     end
419:   end
