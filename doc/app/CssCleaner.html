For use in ActiveRecord models We parse and clean the CSS line by line in order to provide more helpful error messages. The prefix is used if you want to make sure a particular prefix appears on all the selectors in this block of css, eg “.userstuff p” instead of just “p“

36:   def clean_css_code(css_code, prefix = "")
37:     return "" if !css_code.match(/\w/) # only spaces of various kinds
38:     clean_css = ""
39:     parser = CssParser::Parser.new
40:     parser.add_block!(css_code)
41:     prefix = prefix + " " unless prefix.blank?
42:     
43:     if parser.to_s.blank?
44:       errors.add(:base, ts("We couldn't find any valid CSS rules in that code."))
45:     else
46:       parser.each_rule_set do |rs|
47:         selectors = rs.selectors.map do |selector|
48:           if selector.match(/@font-face/)
49:             errors.add(:base, ts("We don't allow the @font-face feature."))
50:             next
51:           end
52:           sel = selector.gsub(/\n/, '').strip
53:           sel = sel.match(/^\s*#{prefix}/) ? sel : prefix + sel
54:         end
55:         clean_declarations = ""
56:         rs.each_declaration do |property, value, is_important|
57:           if property.blank? || value.blank?
58:             errors.add(:base, ts("The code for #{rs.selectors.join(',')} doesn't seem to be a valid CSS rule."))
59:           elsif sanitize_css_property(property).blank?
60:             errors.add(:base, ts("We don't currently allow the CSS property #{property} -- please notify support if you think this is an error."))
61:           elsif (cleanval = sanitize_css_declaration_value(property, value)).blank?
62:             errors.add(:base, ts("The #{property} property in #{rs.selectors.join(', ')} cannot have the value #{value}, sorry!"))
63:           else
64:             clean_declarations += "  #{property}: #{cleanval}#{is_important ? ' !important' : ''};\n"
65:           end
66:         end
67:         if clean_declarations.blank?
68:           errors.add(:base, ts("There don't seem to be any rules for #{rs.selectors.join(',')}"))
69:         else
70:           # everything looks ok, add it to the css

sanitize_css_declaration_value(property, value) click to toggle source

A declaration must match the format: property: value; All properties must appear in ArchiveConfig.SUPPORTED_CSS_PROPERTIES or ArchiveConfig.SUPPORTED_CSS_SHORTHAND_PROPERTIES, or that property and its value will be omitted. All values are sanitized. If any values in a declaration are invalid, the value will be blanked out and an 

  empty property returned.
     # File lib/css_cleaner.rb, line 94
 94:   def sanitize_css_declaration_value(property, value)
 95:     clean = ""
 96:     property.downcase!
 97:     if property == "font-family"
 98:       if !sanitize_css_font(value).blank?
 99:         # preserve the original capitalization
100:         clean = value
101:       end
102:     elsif property == "content"
103:       clean = sanitize_css_content(value)
104:     elsif value.match(/\burl\b/) && (!ArchiveConfig.SUPPORTED_CSS_KEYWORDS.include?("url") || !%(background background-image border border-image list-style list-style-image).include?(property))
105:       # check whether we can use urls in this property
106:       clean = ""
107:     elsif property.match(/#{ArchiveConfig.SUPPORTED_CSS_SHORTHAND_PROPERTIES.join('|')}/)
108:       clean = tokenize_and_sanitize_css_value(value)
109:     elsif ArchiveConfig.SUPPORTED_CSS_PROPERTIES.include?(property)
110:       clean = sanitize_css_value(value)
111:     end
112:     clean.strip
113:   end

116:   def tokenize_and_sanitize_css_value(value)
117:     cleanval = ""
118:     scanner = StringScanner.new(value)
119:     
120:     # we scan until we find either a space, a comma, or an open parenthesis
121:     while scanner.exist?(/\s+|,|\(/)
122:       # we have some tokens left to break up
123:       in_paren = 0
124:       token = scanner.scan_until(/\s+|,|\(/)
125:       if token.blank? || token == ","
126:         cleanval += token
127:         next
128:       end
129:       in_paren = 1 if token.match(/\($/)
130:       while in_paren > 0
131:         # scan until closing paren or another opening paren
132:         nextpart = scanner.scan_until(/\(|\)/)
133:         if nextpart
134:           token += nextpart
135:           in_paren += 1 if token.match(/\($/)
136:           in_paren -= 1 if token.match(/\)$/)
137:         else
138:           # mismatched parens
139:           return ""
140:         end
141:       end
142:       
143:       # we now have a single token
144:       separator = token.match(/(\s|,)$/) || ""
145:       token.strip!
