 65:   def show
 66:     # don't bother collecting tags unless the user gets to see them
 67:     if @tag_set.visible || @tag_set.user_is_moderator?(current_user)
 68: 
 69:       # we use this to collect up fandom parents for characters and relationships
 70:       @fandom_keys_from_other_tags = []
 71:       
 72:       # we use this to store the tag name results
 73:       @tag_hash = HashWithIndifferentAccess.new
 74:       
 75:       %(character relationship).each do |tag_type|
 76:         if @tag_set.has_type?(tag_type)
 77:           ## names_by_parent returns a hash of arrays like so: 
 78:           ## hash[parent_name] => [child name, child name, child name] 
 79: 
 80:           # get the manually associated fandoms
 81:           assoc_hash = TagSetAssociation.names_by_parent(TagSetAssociation.for_tag_set(@tag_set), tag_type)
 82:           
 83:           # get canonically associated fandoms
 84:           canonical_hash = Tag.names_by_parent(tag_type.classify.constantize.in_tag_set(@tag_set), "fandom")
 85: 
 86:           # merge the values of the two hashes (each value is an array) as a set (ie remove duplicates)
 87:           @tag_hash[tag_type] = assoc_hash.merge(canonical_hash) {|key, oldval, newval| (oldval | newval) }
 88: 
 89:           # get any tags without a fandom
 90:           remaining = @tag_set.with_type(tag_type).with_no_parents
 91:           if remaining.count > 0
 92:             @tag_hash[tag_type]["(No linked fandom - might need association)"] ||= []
 93:             @tag_hash[tag_type]["(No linked fandom - might need association)"] += remaining.value_of(:name)
 94:           end
 95: 
 96:           # store the parents 
 97:           @fandom_keys_from_other_tags += @tag_hash[tag_type].keys
 98:         end
 99:       end
100:       
101:       # get rid of duplicates and sort
102:       @fandom_keys_from_other_tags = @fandom_keys_from_other_tags.compact.uniq.sort {|a,b| a.gsub(/^(the |an |a )/, '') <=> b.gsub(/^(the |an |a )/, '')}       
103:       
104:       # now handle fandoms
105:       if @tag_set.has_type?("fandom")
106:         # Get fandoms hashed by media -- just the canonical associations
107:         @tag_hash[:fandom] = Tag.names_by_parent(Fandom.in_tag_set(@tag_set), "media") 
108:         
109:         # get any fandoms without a media
110:         if @tag_set.with_type("fandom").with_no_parents.count > 0
111:           @tag_hash[:fandom]["(No Media)"] ||= []
112:           @tag_hash[:fandom]["(No Media)"] += @tag_set.with_type("fandom").with_no_parents.value_of(:name)
113:         end
114: 
115:         # we want to collect and warn about any chars or relationships not in the set's fandoms
116:         @character_seen = {}
117:         @relationship_seen = {}
118: 
119:         # clear out the fandoms we're showing from the list of other tags
120:         if @fandom_keys_from_other_tags && !@fandom_keys_from_other_tags.empty?
121:           @fandom_keys_from_other_tags -= @tag_hash[:fandom].values.flatten
122:         end
123: 
124:         @unassociated_chars = []
125:         @unassociated_rels = []
126:         unless @fandom_keys_from_other_tags.empty?
127:           if @tag_hash[:character]
128:             @unassociated_chars = @tag_hash[:character].values_at(*@fandom_keys_from_other_tags).flatten.compact.uniq
129:           end
