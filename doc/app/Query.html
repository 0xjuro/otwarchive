145:   end

 76:   def Query.split_query(query={})
 77:     with = {}
 78:     errors = []
 79:     text = query[:text] || ""
 80:     # transform
 81:     #   (field1,field2): search string
 82:     # into sphinx's multiple-field search operator
 83:     #   @(field1,field2) search string
 84:     match = text.match(/(\(\S+?\,\S+?\)):( .+)$/)
 85:     if match
 86:       text = match.pre_match + match.post_match
 87:       text = text + "@" + match[1]  + match[2]
 88:     end
 89:     for string in ALL_FIELDS
 90:       text = (text + " @" + string + " " + query[string.to_sym]) unless query[string.to_sym].blank?
 91:     end
 92:     text = (text + " @type " + query[:type]) unless query[:type].blank?
 93:     for string in %{word hit kudo bookmark} do
 94:       sym = string.pluralize.to_sym
 95:       unless query[sym].blank?
 96:         match = query[sym].match(/^([<>]*)\s*([\d,. -]+)\s*$/)
 97:         if match
 98:           with[(string + "_count").to_sym] = Query.numerical_range(match[1], match[2])
 99:         else

101:         end
102:       end
103:     end
104:     with[:rec] = true if query[:rec]

106:     with[:recced] = true if query[:recced]

108:     with[:bookmarker] = Range.new(1,1000000) if query[:bookmarked]

standardize(query) click to toggle source

this is used to standardize a query, specifically, moving things like “author: astolat words: >1,000” to :author => {“astolat”}, :words => {“> 1000”} 

    # File lib/query.rb, line 37
37:   def Query.standardize(query)
38:     Rails.logger.debug "original query: #{query.to_s}"
39:     query[:text] = "" unless query[:text]
40:     # change something: to @something so we know a section ends when the next section starts
41:     for string in ALL_INDEXES
42:       query[:text] = query[:text].sub(/#{string}:/, "@#{string} ")
43:     end
44:     # remove a single multiple-field search operator of the form (i.e. leave it in :text)
45:     #   (field1,field2): search string
46:     match = query[:text].match(/\(\S+?\,\S+?\): ([^@]*)/)
47:     if match
48:       query[:group] = match[0]
49:       query[:text] = match.pre_match + match.post_match
50:     end
51:     for string in ALL_INDEXES
52:       match = query[:text].match(/@#{string} ([^@]*)/)
53:       if match
54:         query[:text] = match.pre_match + match.post_match
55:         query[string.to_sym] = match[1]
56:       end
57:     end
58:     # add multiple-field search operator back at the end
59:     if query[:group]
60:       query[:text] = query[:text] + query.delete(:group)
61:     end
62:     query.each { |k, v| query[k] = v.strip }
63:     query.delete_if { |k, v| v.blank? }
64:     # in rails 3, a query with < or > will get encoded, unencode them again
65:     # also, remove punctuation such as , and . (10.000 == 10,000 == 10000)
66:     for string in %{word hit kudo bookmark date} do
67:       sym = string.pluralize.to_sym unless string == "date"
68:       sym = string.to_sym if string == "date"
69:       query[sym] = query[sym].gsub("&gt;", ">").gsub("&lt;", "<").gsub(/[,.]/, "") if query[sym]
70:     end
71:     return query
72:   end
