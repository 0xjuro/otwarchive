156:     @db_hash = StatHash.new
157:     @render_hash = StatHash.new if $render
158:     @total_hash = StatHash.new
159:     action = key = nil
160:     while @input.gets
161:       if $_.index("Processing ")==0
162:         action = $_.split[1]
163:         next
164:       end
165:       next unless $_.index("Completed in")==0
166:       #get the action unless we are forcing url tracking
167:       if @force_url_use
168:         #the below regexp turns "[http://archiveofourown.org/en/works/5/chapters]" to "/works/5"
169:         key = $_[/\[\S+\]/].gsub(/\S+\/\/(\w|\.)*\/?(\w)*/,'')[/\/\w*\/?\w*\/?\w*/] || '/'
170:       else
171:         key = action
172:       end
173:       db_time = $_[/DB: \d+\.\d+/][/\d+\.\d+/].to_f
174:       @db_hash.add(key, db_time)
175:       render_time = $_[/Rendering: \d+\.\d+/] if $render
