66:   def download_mobi
67:      cmd_pre = %{cd "#{@work.download_dir}"; html2mobi }
68:      # double quotes in title need to be escaped
69:      title = @work.title.gsub(/"/, '\"')
70:      cmd_post = %{ --mobifile "#{@work.download_title}.mobi" --title "#{title}" --author "#{@work.display_authors}" }
71: 
72:     # if only one chapter can use same file as html and pdf versions
73:     if @chapters.size == 1
74:       create_work_html
75: 
76:       # except mobi requires latin1 encoding
77:       unless File.exists?("#{@work.download_dir}/mobi.html")
78:         html = Iconv.conv("LATIN1//TRANSLIT//IGNORE", "UTF8",
79:                  File.read("#{@work.download_basename}.html")).force_encoding("ISO-8859-1")
80:         File.open("#{@work.download_dir}/mobi.html", 'w') {|f| f.write(html)}
81:       end
82: 
83:       # convert latin html to mobi
84:       cmd = cmd_pre + "mobi.html" + cmd_post
85:     else
86:       # more than one chapter
87:       # create a table of contents out of separate chapter files
88:       mobi_files = create_mobi_html
89:       cmd = cmd_pre + mobi_files + " --gentoc" + cmd_post
90:     end
91:     Rails.logger.debug cmd
92:     `#{cmd} 2> /dev/null`
93:     check_for_file("mobi")
94:     send_file("#{@work.download_basename}.mobi", :type => "application/mobi")
95:   end
