111:     keys_to_lookup += fandoms.map {|fandom| "autocomplete_fandom_#{fandom}_#{tag_type}"}.flatten
112:     return [] if keys_to_lookup.empty?
113: 
114:     # if we don't want tags that aren't in the tag set(s), we need to first

116:     # of those tags, and the associated tags
117:     if options[:include_wrangled] == "false"
118:       combo_key2 = combo_key + "2"
119:       combo_key3 = combo_key + "3"
120:       keys_for_intersect = tag_sets.map {|set| "autocomplete_tagset_#{tag_type}_#{set}"}.flatten
121:       $redis.zunionstore(combo_key2, keys_to_lookup, :aggregate => :max)
122:       $redis.zunionstore(combo_key3, keys_for_intersect, :aggregate => :max)
