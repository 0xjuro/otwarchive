214:   def load_batch_associations!(batch_associations, options = {})
215:     options.reverse_merge!({:do_relationships => false})
216:     association_lines = batch_associations.split("\n")
217:     fandom_tagnames_to_add = []
218:     child_tagnames_to_add = []
219:     assocs_to_save = []
220:     failed = []
221:     canonical = []
222:     
223:     association_lines.each do |line|
224:       children_names = line.split(',')
225:       parent_name = children_names.shift.strip
226:       parent_tag_id = Fandom.where(:name => parent_name).value_of(:id).first
227:       unless parent_tag_id
228:         failed << line
229:         next
230:       end
231:       failed_children = []
232:       added_parent = false
233:       children_names.map {|c| c.strip}.each_with_index do |child_name|
234:         child_tag_id = (options[:do_relationships] ? Relationship : Character).where(:name => child_name).value_of(:id).first
235:         unless child_tag_id
236:           failed_children << child_name
237:           next
238:         end
239:         assoc = tag_set_associations.build(:tag_id => child_tag_id, :parent_tag_id => parent_tag_id)
240:         unless assoc.valid?
241:           failed_children << child_name
242:           next
243:         end
244:         assocs_to_save << assoc
245:         child_tagnames_to_add << child_name
246:         fandom_tagnames_to_add << parent_name unless added_parent
247:         added_parent = true
248:       end
249:       failed << "#{parent_name},#{failed_children.join(',')}" unless failed_children.empty?
250:     end
251:     
252:     # add the tags to the set
253:     tag_set.fandom_tagnames_to_add = fandom_tagnames_to_add
254:     tag_set.send (options[:do_relationships] ? :relationship_tagnames_to_add= : :character_tagnames_to_add=), child_tagnames_to_add
255:     
256:     if tag_set.save
257:       # save the associations
258:       assocs_to_save.each {|assoc| assoc.save}
259:     else
260:       # whoops, nothing worked
261:       failed = association_lines
262:     end
263:         
264:     return failed
265:   end
