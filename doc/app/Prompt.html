 76:     prompt_type = self.class.name
 77:     restriction = get_prompt_restriction
 78:     if restriction
 79:       # make sure tagset has no more/less than the required/allowed number of tags of each type
 80:       TagSet::TAG_TYPES.each do |tag_type|
 81:         # get the tags of this type the user has specified
 82:         taglist = tag_set ? eval("tag_set.#{tag_type}_taglist") : []        
 83:         tag_count = taglist.count
 84: 
 85:         # check if user has chosen the "Any" option
 86:         if self.send("any_#{tag_type}")
 87:           if tag_count > 0
 88:             errors.add(:base, ts("^You have specified tags for %{tag_type} in your %{prompt_type} but also chose 'Any,' which will override them! Please only choose one or the other.", 
 89:                                 :tag_type => tag_type, :prompt_type => prompt_type))
 90:           end
 91:           next
 92:         end
 93: 
 94:         # otherwise let's make sure they offered the right number of tags
 95:         required = eval("restriction.#{tag_type}_num_required")
 96:         allowed = eval("restriction.#{tag_type}_num_allowed")
 97:         unless tag_count.between?(required, allowed)
 98:           taglist_string = taglist.empty? ?
 99:               ts("none") :
100:               "(#{tag_count}) -- " + taglist.collect(&:name).join(ArchiveConfig.DELIMITER_FOR_OUTPUT)
101:           if allowed == 0
102:             errors.add(:base, ts("^#{prompt_type} cannot include any #{tag_type} tags. You currently have %{taglist}.",
103:                                  :taglist => taglist_string))
104:           elsif required == allowed
105:             errors.add(:base, ts("^#{prompt_type} must have exactly %{required} #{tag_type} tags. You currently have %{taglist}.",
106:               :required => required, :taglist => taglist_string))
107:           else
108:             errors.add(:base, ts("^#{prompt_type} must have between %{required} and %{allowed} #{tag_type} tags. You currently have %{taglist}.",

     # File app/models/prompt.rb, line 217
217:   def match(other, settings=nil)
218:     return nil if settings.nil?
219:     potential_prompt_match_attributes = {:offer => other, :request => self}
220:     full_request_tag_set = self.optional_tag_set ? self.tag_set + self.optional_tag_set : self.tag_set
221:     full_offer_tag_set = other.optional_tag_set ? (other.tag_set + other.optional_tag_set) : other.tag_set
222: 
223:     TagSet::TAG_TYPES.each do |type|
224:       if self.send("any_#{type}") || other.send("any_#{type}")
225:         match_count = ALL
226:       else
227:         required_count = settings.send("num_required_#{type.pluralize}")
228:         if settings.send("include_optional_#{type.pluralize}")
229:           match_count = full_request_tag_set.match_rank(full_offer_tag_set, type)
230:         else
231:           # we don't use optional tags to count towards required
232:           match_count = self.tag_set.match_rank(other.tag_set, type)
233:         end
234:       
235:         # if we have to match all and don't, not a match
236:         return nil if required_count == ALL && match_count != ALL
237: 
238:         # we are a match only if we either match all or at least as many as required
239:         return nil if match_count != ALL && match_count < required_count
240: 
241:         # now get the match rank including optional tags if we didn't before
242:         if !settings.send("include_optional_#{type.pluralize}")
243:           match_count = full_request_tag_set.match_rank(full_offer_tag_set, type)
244:         end
245:       end
246: 
247:       potential_prompt_match_attributes["num_#{type.pluralize}_matched"] = match_count
248:     end
249:     return PotentialPromptMatch.new(potential_prompt_match_attributes)
250:   end
