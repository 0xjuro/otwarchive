 87:   def approve_automatically
 88:     if self.new_record?
 89:       # approve with the current user, who is the person who has just
 90:       # added this item -- might be either moderator or owner
 91:       approve(User.current_user == :false ? nil : User.current_user)
 92: 
 93:       # if the collection is open or the user who owns this work is a member, go ahead and approve
 94:       # for the collection
 95:       if !approved_by_collection? && collection
 96:         if !collection.moderated? || collection.user_is_maintainer?(User.current_user) || collection.user_is_posting_participant?(User.current_user)
 97:           approve_by_collection
 98:         end
 99:       end
100: 
101:       # if at least one of the owners of the items automatically approves
102:       # adding or is a member of the collection, go ahead and approve by user
103:       if !approved_by_user?
104:         case item_type
105:         when "Work"
106:           users = item.users || [User.current_user] # if the work has no users, it is also new and being created by the current user
107:         when "Bookmark"
108:           users = [item.user] || [User.current_user]
109:         end
110: 
111:         users.each do |user|
112:           if user.preference.automatically_approve_collections || (collection && collection.user_is_posting_participant?(user))

115:           end

117:       end
