add_paragraphs_to_text(text) click to toggle source

Simplified parser/formatter steps:

  1. Convert newlines into paragraph/break tags based on some simple rules

  2. Parse document with Nokogiri and export xhtml to get pretty-printed and well-formed (not necessarily validating!) xhtml with all tags closed. 

     # File lib/html_cleaner.rb, line 137
137:   def add_paragraphs_to_text(text)
138: 
139:     # get rid of spaces and newlines-before/after-paragraphs and linebreaks
140:     # this enables us to avoid converting newlines into paras/breaks where we already have them
141:     source = text.gsub(/\s*(<p[^>]*>)\s*/, '\1')   # replace all whitespace before/after <p>
142:     source.gsub!(/\s*(<\/p>)\s*/, '\1')            # replace all whitespace before/after </p>
143:     source.gsub!(/\s*(<br\s*?\/?>)\s*/, '<br />')  # replace all whitespace before/after <br>  
144: 
145:     # do we have a paragraph to start and end
146:     source = '<p>' + source unless source.match(/^<p/)
147:     source = source + "</p>" unless source.match(/<\/p>$/)
148:     
149:     # If we have three newlines, assume user wants a blank line
150:     source.gsub!(/\n\s*?\n\s*?\n/, "\n\n&nbsp;\n\n")
151: 
152:     # Convert double newlines into single paragraph break
153:     source.gsub!(/\n+\s*?\n+/, '</p><p>')
154: 
155:     # Convert single newlines into br tags
156:     source.gsub!(/\n/, '<br />')
157:     
158:     # convert double br tags into p tags
159:     source.gsub!(/<br\s*?\/?>\s*<br\s*?\/?>/, '</p><p>')
160:     
161:     # if we have closed inline tags that cross a <p> tag, reopen them 
162:     # at the start of each paragraph before the end
163:     HTML_TAGS_TO_REOPEN.each do |tag|      
164:       source.gsub!(/(<#{tag}>)(.*?)(<\/#{tag}>)/) { $1 + reopen_tags($2, tag) + $3 }
165:     end
166:     
167:     # reopen paragraph tags that cross a <div> tag
168:     source.gsub!(/(<p[^>]*>)(.*?)(<\/p>)/) { $1 + reopen_tags($2, "p", "div") + $3 }
169:     
170:     # swap order of paragraphs around divs
171:     source.gsub!(/(<p[^>]*>)(<div[^>]*>)/, '\2\1')
172: 
173:     # Parse in Nokogiri
174:     parsed = Nokogiri::HTML.parse(source)
175:     parsed.encoding = 'UTF-8'
176:     
177:     # Get out the nice well-formed XHTML
178:     source = parsed.css("body").to_xhtml
179:     
180:     # trash empty paragraphs and leading spaces
181:     source.gsub!(/\s*<p[^>]*>\s*<\/p>\s*/, "")
182:     source.gsub!(/^\s*/, '')
183:     
184:     # get rid of the newlines-before/after-paragraphs inserted by to_xhtml,
185:     # so that when this is loaded up by strip_html_breaks in textarea fields,
186:     # 
187:     source.gsub!(/\s*(<p[^>]*>)\s*/, '\1')
188:     source.gsub!(/\s*(<\/p>)\s*/, '\1')
189:     
190:     # trash the body tag
191:     source.gsub!(/<\/?body>\s*/, '')
192:     
193:     # return the text
194:     source
195:   end

sanitize_field(object, fieldname) click to toggle source

If we aren’t sure that this field hasn’t been sanitized since the last sanitizer version, we sanitize it before we allow it to pass through (and save it if possible). 

    # File lib/html_cleaner.rb, line 6
 6:   def sanitize_field(object, fieldname)
 7:     return "" if object.send(fieldname).nil?
 8:     if object.respond_to?("#{fieldname}_sanitizer_version")
 9:       if object.send("#{fieldname}_sanitizer_version") < ArchiveConfig.SANITIZER_VERSION
10:         # sanitize and save it
11:         Rails.logger.debug "Sanitizing and saving #{fieldname} on #{object.class.name} (id #{object.id})"
12:         object.update_attribute(fieldname, sanitize_value(fieldname, object.send("#{fieldname}")))
13:         object.update_attribute("#{fieldname}_sanitizer_version", ArchiveConfig.SANITIZER_VERSION)
14:       end
15:       # return the field without sanitizing
16:       Rails.logger.debug "Already sanitized #{fieldname} on #{object.class.name} (id #{object.id})"
17:       object.send("#{fieldname}")
18:     else
19:       # no sanitizer version information, so re-sanitize 
20:       Rails.logger.debug "Sanitizing without saving #{fieldname} on #{object.class.name} (id #{object.id})"
21:       sanitize_value(fieldname, object.send("#{fieldname}"))
22:     end
23:   end

63:     if ArchiveConfig.NONZERO_INTEGER_PARAMETERS.has_key?(field.to_s)
64:       return (value.to_i > 0) ? value.to_i : ArchiveConfig.NONZERO_INTEGER_PARAMETERS[field.to_s]
65:     end
66:     return "" if value.blank?
67:     value.strip!
68:     if field.to_s == 'title'
69:       # prevent invisible titles
70:       value.gsub!("<", "&lt;")
71:       value.gsub!(">", "&gt;")
72:     end
73:     if ArchiveConfig.FIELDS_ALLOWING_LESS_THAN.include?(field.to_s)
74:       value.gsub!("<", "&lt;")
75:     end
76:     if ArchiveConfig.FIELDS_ALLOWING_HTML.include?(field.to_s)
77:       # We're allowing users to use HTML in this field
78:       transformers = []
79:       if ArchiveConfig.FIELDS_ALLOWING_VIDEO_EMBEDS.include?(field.to_s)
80:         transformers << Sanitize::Transformers::ALLOW_VIDEO_EMBEDS 
81:       end
82:       if ArchiveConfig.FIELDS_ALLOWING_CSS.include?(field.to_s)
83:         transformers << Sanitize::Transformers::ALLOW_USER_CLASSES
84:       end   
85:       value = Sanitize.clean(add_paragraphs_to_text(fix_bad_characters(value)), 
86:                              Sanitize::Config::ARCHIVE.merge(:transformers => transformers))
87:     else
88:       # clean out all tags
89:       value = Sanitize.clean(fix_bad_characters(value))
90:     end

92:     # Plain text fields can't contain &amp; entities:
93:     value.gsub!(/&amp;/, '&') unless (ArchiveConfig.FIELDS_ALLOWING_HTML_ENTITIES + ArchiveConfig.FIELDS_ALLOWING_HTML).include?(field.to_s)
