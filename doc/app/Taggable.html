parse_tags(klass, incoming_tags) click to toggle source

Process a string or array of tags from any tag class 

     # File lib/taggable.rb, line 172
172:   def parse_tags(klass, incoming_tags)
173:     tags = []
174:     self.invalid_tags ||= []
175:     klass_symbol = klass.to_s.downcase.pluralize.to_sym
176:     tag_array = incoming_tags.is_a?(String) ? incoming_tags.split(ArchiveConfig.DELIMITER_FOR_INPUT) : incoming_tags
177:     tag_array.each do |string|
178:       string.strip!
179:       unless string.blank?
180:         tag = klass.find_or_create_by_name(string)
181:         if tag.valid?
182:           tags << tag if tag.is_a?(klass)
183:         else
184:           self.invalid_tags << tag
185:         end
186:       end
187:     end
188:     if self.preview_mode
189:       self.placeholder_tags ||= {}
190:       self.placeholder_tags[klass_symbol] = tags.uniq
191:     else
192:       # we have to destroy the taggings directly in order to trigger the callbacks
193:       remove = self.send(klass_symbol) - tags
194:       remove.each do |tag|
195:         tagging = Tagging.find_by_tag(self, tag)
196:         tagging.destroy if tagging
197:       end
198:       self.send(klass_symbol.to_s + '=', tags.uniq)
199:     end
200:   end
